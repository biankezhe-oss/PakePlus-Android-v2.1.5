    // 获取DOM元素

    const totalPriceEl = document.getElementById('totalPrice');

    // 当前选中的分类
    let currentCategory = 'pneumatic';

    // 初始化弹簧数量下拉框
    function initSpringCountSelect(model) {
      springCountSelect.innerHTML = '';
      const maxCount = productData.pneumatic.springMaxCount[model] || 12;
      for (let i = 1; i <= maxCount; i++) {
        const option = document.createElement('option');
        option.value = i;
        option.textContent = i;
        if (i === 1) option.selected = true;
        springCountSelect.appendChild(option);
      }
    }

    // 初始化型号下拉框
    function initModelSelect(category) {
      modelSelect.innerHTML = '';
      const models = Object.keys(productData[category].bodyPrice.ss304);
      models.forEach(model => {
        const option = document.createElement('option');
        option.value = model;
        option.textContent = model;
        if (model === (category === 'pneumatic' ? 'KZ-45' : 'HD120')) {
          option.selected = true;
        }
        modelSelect.appendChild(option);
      });

      if (category === 'hydraulic') {
        typeGroup.classList.add('hidden');
        springCountGroup.classList.add('hidden');
      } else {
        typeGroup.classList.remove('hidden');
        if (typeSelect.value === 'double') {
          springCountGroup.classList.add('hidden');
        } else {
          initSpringCountSelect(modelSelect.value);
          springCountGroup.classList.remove('hidden');
        }
      }
    }

    // 价格更新动画函数（科技风数字滚动）
    function updatePriceWithAnimation(finalPrice) {
      // 移除加载类，添加动画类
      totalPriceEl.classList.remove('loading');
      totalPriceEl.classList.add('price-update');
      // 更新价格
      totalPriceEl.textContent = `¥${finalPrice} 元`;
      // 移除动画类
      setTimeout(() => {
        totalPriceEl.classList.remove('price-update');
      }, 600);
    }

    // 计算价格
    function calculatePrice() {
      // 计算中状态（科技风加载动效）
      totalPriceEl.textContent = '计算中...';
      totalPriceEl.classList.add('loading');
      
      setTimeout(() => {
        const material = materialSelect.value;
        const model = modelSelect.value;
        const type = currentCategory === 'pneumatic' ? typeSelect.value : 'double';
        const springCount = parseInt(springCountSelect.value) || 1;
        const discount = parseFloat(discountSelect.value);

        let basePrice;
        if (material === 'ss316l') {
          basePrice = Math.round(productData[currentCategory].bodyPrice.ss316[model] * 1.05);
        } else {
          basePrice = productData[currentCategory].bodyPrice[material][model];
        }

        let totalBasePrice = basePrice;
        if (currentCategory === 'pneumatic' && type === 'single') {
          const springUnitPrice = productData[currentCategory].springUnitPrice[model];
          totalBasePrice += Math.round(springCount * springUnitPrice);
        }

        const finalPrice = Math.round(totalBasePrice * discount);
        updatePriceWithAnimation(finalPrice);
      }, 150);
    }

    // 防抖函数（优化性能）
    function debounce(fn, delay = 150) {
      let timer = null;
      return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => {
          fn.apply(this, args);
        }, delay);
      };
    }

        initModelSelect(currentCategory);
        calculatePrice();

        }

      }
    });
